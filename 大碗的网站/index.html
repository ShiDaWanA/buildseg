// 获取文本框元素 var textBox = document.getElementById('yourTextBoxId'); // 请替换成你的文本框的实际 ID // 检查是否有本地存储的文本数据 if (localStorage.getItem('savedText')) { // 如果有本地存储的文本数据，将其设置为文本框的值 textBox.value = localStorage.getItem('savedText'); } // 监听文本框内容变化事件 textBox.addEventListener('input', function () { // 将文本框的内容保存到本地存储 localStorage.setItem('savedText', textBox.value); }); // 清除本地存储中的文本数据（例如，当点击“删除”按钮时） function clearSavedText() { localStorage.removeItem('savedText'); }
